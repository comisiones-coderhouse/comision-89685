export/import :

- Export default -> import sin llaves : 
- Export comun -> import con llaves : 

    ej comun : 
        //archivo1.jsx
        export const nombre = "Horacio";

        //archivo2.jsx
        import {nombre} from "./archivo1";


    ej default : 
        //archivo1.jsx
        const nombre = "Horacio";
        export default nombre;

        //archivo2.jsx
        import nombre from "./archivo1";
